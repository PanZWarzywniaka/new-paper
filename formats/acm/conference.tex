% Define document class
\documentclass[sigconf]{acmart}

% Pre-abstract formatting
\newcommand{\preabstract}{

    % Set the title
    \ifcsname shortpapertitle\endcsname
        \title[\shortpapertitle]{\papertitle}
    \else
        \title{\papertitle}
    \fi

    % Define the author block
    \ifcsname doubleblind\endcsname
        \author{\doubleblind}
    \else
        \ifcsname firstauthor\endcsname
            \author{\firstauthor}
            \affiliation{
                \institution{\firstauthoraffiliation}
                \country{\firstauthorcountry}
            }
        \fi
        \ifcsname collaboratorone\endcsname
            \author{\collaboratorone}
            \affiliation{
                \institution{\collaboratoroneaffiliation}
                \country{\collaboratoronecountry}
            }
        \fi
        \ifcsname collaboratortwo\endcsname
            \author{\collaboratortwo}
            \affiliation{
                \institution{\collaboratortwoaffiliation}
                \country{\collaboratortwocountry}
            }
        \fi
        \ifcsname collaboratorthree\endcsname
            \author{\collaboratorthree}
            \affiliation{
                \institution{\collaboratorthreeaffiliation}
                \country{\collaboratorthreecountry}
            }
        \fi
        \ifcsname lastauthor\endcsname
            \author{\lastauthor}
            \affiliation{
                \institution{\lastauthoraffiliation}
                \country{\lastauthorcountry}
            }
        \fi
    \fi

    % Set the bibliography style
    \bibliographystyle{ACM-Reference-Format}
}

% Post-abstract formatting
\newcommand{\postabstract}{

    % Title is formatted after the abstract for this style
    \maketitle
}
